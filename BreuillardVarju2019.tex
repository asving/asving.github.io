\subsection{Statement of main results}

Let $\mu_{\lambda}$ be the Bernoulli convolution of parameter $\lambda \in (0,1)$. Denote by $\mathcal{P}_d$ the set of polynomials of degree at most $d$ all of whose coefficients are $-1, 0$ or $1$. We write $$E_{d, \alpha} = \{ \eta \in (1/2,1) \,:\, \dim \mu_{\eta} < \alpha \text{ and } P(\eta) = 0 \text{ for some } P \in \mathcal{P}_{d} \}. $$

\begin{theorem}
    Let $\lambda \in (1/2,1)$ such that $\dim \mu_{\lambda} < 1$. Then for every $\eps > 0$ there is a number $A > 0$ such that the following holds. For every sufficiently large integer $d_0$ there is an integer $$d\in [d_0, \exp^{(5)}(\log^{(5)}(d_0)  + A )]$$ and $\eta \in E_{d, \dim \mu_{\lambda} + \eps}$ such that $$|\lambda - \eta| \leq \exp(-d^{\log^{(3)} d}).$$
\end{theorem}

$\log$ and $\exp$ are in base 2. $\log^{(a)}$ and $\exp^{(a)}$ are the $a$-fold iterations of $\log$ and $\exp$.

\begin{corollary}
    Let $\lambda \in (1/2,1)$ be a number such that $$|P(\lambda)| > \exp(-d^{\log^{(3)}d})$$ for all $P \in \mathcal{P}_d$ for all sufficiently large $d$. Then $\dim \mu_{\lambda} = 1$.
\end{corollary}

\begin{proof}
    Let $P(x) = a_0x^d + a_1 x^{d-1} + \ldots + a_{d-1}x +  a_d$ be in $\mathcal{P}_d$. Then $P'(x) = da_0 + (d-1)a_1 x^{d-2} + \ldots + a_{d-1}$. Hence it follows for all $x\in (0,1)$ that $|P'(x)| < d + (d-1) + \ldots + 1 = \frac{d(d + 1)}{2}$. So if there is a number $\eta$ that is a root of a polynomial $P \in \mathcal{P}_d$ such that $$|\lambda - \eta| \leq \frac{2}{d(d + 1)} \exp(-d^{\log^{(3)}d})$$ then $|P(\lambda)| \leq \exp(-d^{\log^{(3)}d})$. In the theorem above, the factor $\frac{2}{d(d+1)}$ will be insignificant and so the corollay will follow. 
\end{proof}
